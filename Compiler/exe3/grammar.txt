/* Program */
program  → basic id ( ) block
block    → { decls stmts }

/* Declaration */
decls    → decls decl | ε
decl     → type id ;
type     → type [ num ] | basic
basic    → float | int

/* Statement */
stmts    → stmts stmt | ε
stmt     → id = expr ;
           | loc = expr ;
           | if ( bool ) stmt
           | if ( bool ) stmt else stmt
           | while ( bool ) stmt
           | do stmt while ( bool ) ;
           | break ;
           | continue ;
           | block
loc → loc [ num ] | id [ num ]

/* Condition Expressions */
bool     → bool || join | join
join     → join && equality | equality
equality → expr rel expr | true | false
rel      → < | > | == | != | <= | >=

/* Arithmetic Expressions */
expr     → expr1 opa term | term
term     → temr1 opb unary | unary
unary    → opc unary | factor
factor   → ( expr ) | loc | id | num | real
opa      → + | -
opb      → * | /
opc      → ! | -
